let s = "abcabcbb";
let arr = [];
let x = '';
let count = 0;

for (let i=0; i<s.length; i++){
    if(!x.includes(s[i])) {
        x = x + s[i];
    } else {
        arr.push(x);
        x = s[i];
    }
}
for (let i of arr){
    if (i.length > count){
        count = i.length;
    }
    
}
console.log(count);
